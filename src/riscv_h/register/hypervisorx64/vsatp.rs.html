6/// Virtual Supervisor Address Translation and Protection Register.
7#[derive(Copy, Clone, Debug)]
8pub struct Vsatp {
9    bits: usize,
10}
11
12impl Vsatp {
13    /// Returns the raw bits of the register.
14    #[inline]
15    pub fn bits(&self) -> usize {
16        return self.bits;
17    }
18    /// Creates a register value from raw bits.
19    #[inline]
20    pub fn from_bits(x: usize) -> Self {
21        return Vsatp { bits: x };
22    }
23    /// Writes the register value to the CSR.
24    #[inline]
25    pub unsafe fn write(&self) {
26        _write(self.bits);
27    }
28    /// Returns the guest address translation mode.
29    #[inline]
30    pub fn mode(&self) -> HgatpValues {
31        HgatpValues::from(self.bits.get_bits(60..64))
32    }
33    /// Sets the guest address translation mode.
34    #[inline]
35    pub fn set_mode(&mut self, val: HgatpValues) {
36        self.bits.set_bits(60..64, val as usize);
37    }
38    /// Returns the address space identifier.
39    #[inline]
40    pub fn asid(&self) -> usize {
41        self.bits.get_bits(44..60)
42    }
43    /// Sets the address space identifier.
44    #[inline]
45    pub fn set_asid(&mut self, val: usize) {
46        self.bits.set_bits(44..60, val);
47    }
48    /// Returns the physical page number for root page table.
49    #[inline]
50    pub fn ppn(&self) -> usize {
51        self.bits.get_bits(0..44)
52    }
53    /// Sets the physical page number for root page table.
54    #[inline]
55    pub fn set_ppn(&mut self, val: usize) {
56        self.bits.set_bits(0..44, val);
57    }
58}

66/// Hypervisor Guest Address Translation and Protection Register values.
67#[derive(Copy, Clone, Debug)]
68#[repr(usize)]
69pub enum HgatpValues {
70    /// Bare
71    Bare = 0,
72    /// Supervisor Virtual Address Translation (SV39)
73    Sv39x4 = 8,
74    /// Supervisor Virtual Address Translation (SV48)
75    Sv48x4 = 9,
76}
77
78impl HgatpValues {
79    fn from(x: usize) -> Self {
80        match x {
81            0 => Self::Bare,
82            8 => Self::Sv39x4,
83            9 => Self::Sv48x4,
84            _ => unreachable!(),
85        }
86    }
87}
